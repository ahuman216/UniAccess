const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; const recognition = new SpeechRecognition(); recognition.continuous = true; recognition.onresult = e => { document.getElementById('output').innerText += e.results[e.results.length-1][0].transcript + ' '; }; document.getElementById('start').onclick = () => recognition.start();
